208:         return @stack.shift if @stack.size > 0
209:         #STDERR.puts @source.encoding
210:         @source.read if @source.buffer.size<2
211:         #STDERR.puts "BUFFER = #{@source.buffer.inspect}"
212:         if @document_status == nil

214:           word = @source.match( /^((?:\s+)|(?:<[^>]*>))/um )
215:           word = word[1] unless word.nil?
216:           #STDERR.puts "WORD = #{word.inspect}"
217:           case word
218:           when COMMENT_START
219:             return [ :comment, @source.match( COMMENT_PATTERN, true )[1] ]
220:           when XMLDECL_START
221:             #STDERR.puts "XMLDECL"
222:             results = @source.match( XMLDECL_PATTERN, true )[1]
223:             version = VERSION.match( results )
224:             version = version[1] unless version.nil?
225:             encoding = ENCODING.match(results)
226:             encoding = encoding[1] unless encoding.nil?
227:             @source.encoding = encoding
228:             standalone = STANDALONE.match(results)
229:             standalone = standalone[1] unless standalone.nil?

231:           when INSTRUCTION_START
232:             return [ :processing_instruction, *@source.match(INSTRUCTION_PATTERN, true)[1,2] ]
233:           when DOCTYPE_START
234:             md = @source.match( DOCTYPE_PATTERN, true )
235:             @nsstack.unshift(curr_ns=Set.new)
236:             identity = md[1]
237:             close = md[2]
238:             identity =~ IDENTITY
239:             name = $1
240:             raise REXML::ParseException.new("DOCTYPE is missing a name") if name.nil?
241:             pub_sys = $2.nil? ? nil : $2.strip
242:             long_name = $4.nil? ? nil : $4.strip
243:             uri = $6.nil? ? nil : $6.strip
244:             args = [ :start_doctype, name, pub_sys, long_name, uri ]
245:             if close == ">"
246:               @document_status = :after_doctype
247:               @source.read if @source.buffer.size<2
248:               md = @source.match(/^\s*/um, true)
249:               @stack << [ :end_doctype ]
250:             else
251:               @document_status = :in_doctype
252:             end
253:             return args
254:           when /^\s+/
255:           else
256:             @document_status = :after_doctype
257:             @source.read if @source.buffer.size<2
258:             md = @source.match(/\s*/um, true)
259:             if @source.encoding == "UTF-8"
260:               if @source.buffer.respond_to? :force_encoding
261:                 @source.buffer.force_encoding(Encoding::UTF_8)
262:               end
263:             end
264:           end
265:         end
266:         if @document_status == :in_doctype
267:           md = @source.match(/\s*(.*?>)/um)
268:           case md[1]
269:           when SYSTEMENTITY 
270:             match = @source.match( SYSTEMENTITY, true )[1]
271:             return [ :externalentity, match ]
272: 
273:           when ELEMENTDECL_START
274:             return [ :elementdecl, @source.match( ELEMENTDECL_PATTERN, true )[1] ]
275: 
276:           when ENTITY_START
277:             match = @source.match( ENTITYDECL, true ).to_a.compact
278:             match[0] = :entitydecl
279:             ref = false
280:             if match[1] == '%'
281:               ref = true
282:               match.delete_at 1
283:             end
284:             # Now we have to sort out what kind of entity reference this is
285:             if match[2] == 'SYSTEM'
286:               # External reference
287:               match[3] = match[3][1..-2] # PUBID
288:               match.delete_at(4) if match.size > 4 # Chop out NDATA decl
289:               # match is [ :entity, name, SYSTEM, pubid(, ndata)? ]
290:             elsif match[2] == 'PUBLIC'
291:               # External reference
292:               match[3] = match[3][1..-2] # PUBID
293:               match[4] = match[4][1..-2] # HREF
294:               # match is [ :entity, name, PUBLIC, pubid, href ]
295:             else
296:               match[2] = match[2][1..-2]
297:               match.pop if match.size == 4
298:               # match is [ :entity, name, value ]
299:             end
300:             match << '%' if ref
301:             return match
302:           when ATTLISTDECL_START
303:             md = @source.match( ATTLISTDECL_PATTERN, true )
304:             raise REXML::ParseException.new( "Bad ATTLIST declaration!", @source ) if md.nil?
305:             element = md[1]
306:             contents = md[0]
307: 
308:             pairs = {}
309:             values = md[0].scan( ATTDEF_RE )
310:             values.each do |attdef|
311:               unless attdef[3] == "#IMPLIED"
312:                 attdef.compact!
313:                 val = attdef[3]
314:                 val = attdef[4] if val == "#FIXED "
315:                 pairs[attdef[0]] = val
316:                 if attdef[0] =~ /^xmlns:(.*)/
317:                   @nsstack[0] << $1
318:                 end
319:               end
320:             end
321:             return [ :attlistdecl, element, pairs, contents ]
322:           when NOTATIONDECL_START
323:             md = nil
324:             if @source.match( PUBLIC )
325:               md = @source.match( PUBLIC, true )
326:               vals = [md[1],md[2],md[4],md[6]]
327:             elsif @source.match( SYSTEM )
328:               md = @source.match( SYSTEM, true )
329:               vals = [md[1],md[2],nil,md[4]]
330:             else
331:               raise REXML::ParseException.new( "error parsing notation: no matching pattern", @source )
332:             end
333:             return [ :notationdecl, *vals ]
334:           when CDATA_END
335:             @document_status = :after_doctype
336:             @source.match( CDATA_END, true )
337:             return [ :end_doctype ]
338:           end
339:         end
340:         begin
341:           if @source.buffer[0] == ?<
342:             if @source.buffer[1] == ?/
343:               @nsstack.shift
344:               last_tag = @tags.pop
345:               #md = @source.match_to_consume( '>', CLOSE_MATCH)
346:               md = @source.match( CLOSE_MATCH, true )
347:               raise REXML::ParseException.new( "Missing end tag for "+
348:                 "'#{last_tag}' (got \"#{md[1]}\")", 
349:                 @source) unless last_tag == md[1]
350:               return [ :end_element, last_tag ]
351:             elsif @source.buffer[1] == ?!
352:               md = @source.match(/\A(\s*[^>]*>)/um)
353:               #STDERR.puts "SOURCE BUFFER = #{source.buffer}, #{source.buffer.size}"
354:               raise REXML::ParseException.new("Malformed node", @source) unless md
355:               if md[0][2] == ?-
356:                 md = @source.match( COMMENT_PATTERN, true )
357: 
358:                 case md[1]
359:                 when /--/, /-$/
360:                   raise REXML::ParseException.new("Malformed comment", @source)
361:                 end
362: 
363:                 return [ :comment, md[1] ] if md
364:               else
365:                 md = @source.match( CDATA_PATTERN, true )
366:                 return [ :cdata, md[1] ] if md
367:               end
368:               raise REXML::ParseException.new( "Declarations can only occur "+
369:                 "in the doctype declaration.", @source)
370:             elsif @source.buffer[1] == ??
371:               md = @source.match( INSTRUCTION_PATTERN, true )
372:               return [ :processing_instruction, md[1], md[2] ] if md
373:               raise REXML::ParseException.new( "Bad instruction declaration",
374:                 @source)
375:             else
376:               # Get the next tag
377:               md = @source.match(TAG_MATCH, true)
378:               unless md
379:                 # Check for missing attribute quotes
380:                 raise REXML::ParseException.new("missing attribute quote", @source) if @source.match(MISSING_ATTRIBUTE_QUOTES )
381:                 raise REXML::ParseException.new("malformed XML: missing tag start", @source) 
382:               end
383:               attributes = {}
384:               prefixes = Set.new
385:               prefixes << md[2] if md[2]
386:               @nsstack.unshift(curr_ns=Set.new)
387:               if md[4].size > 0
388:                 attrs = md[4].scan( ATTRIBUTE_PATTERN )
389:                 raise REXML::ParseException.new( "error parsing attributes: [#{attrs.join ', '}], excess = \"#$'\"", @source) if $' and $'.strip.size > 0
390:                 attrs.each { |a,b,c,d,e| 
391:                   if b == "xmlns"
392:                     if c == "xml"
393:                       if d != "http://www.w3.org/XML/1998/namespace"
394:                         msg = "The 'xml' prefix must not be bound to any other namespace "+
395:                         "(http://www.w3.org/TR/REC-xml-names/#ns-decl)"
396:                         raise REXML::ParseException.new( msg, @source, self )
397:                       end
398:                     elsif c == "xmlns"
399:                       msg = "The 'xmlns' prefix must not be declared "+
400:                       "(http://www.w3.org/TR/REC-xml-names/#ns-decl)"
401:                       raise REXML::ParseException.new( msg, @source, self)
402:                     end
403:                     curr_ns << c
404:                   elsif b
405:                     prefixes << b unless b == "xml"
406:                   end
407: 
408:                   if attributes.has_key? a
409:                     msg = "Duplicate attribute #{a.inspect}"
410:                     raise REXML::ParseException.new( msg, @source, self)
411:                   end
412: 
413:                   attributes[a] = e 
414:                 }
415:               end
416:         
417:               # Verify that all of the prefixes have been defined
418:               for prefix in prefixes
419:                 unless @nsstack.find{|k| k.member?(prefix)}
420:                   raise UndefinedNamespaceException.new(prefix,@source,self)
421:                 end
422:               end
423: 
424:               if md[6]
425:                 @closed = md[1]
426:                 @nsstack.shift
427:               else
428:                 @tags.push( md[1] )
429:               end
430:               return [ :start_element, md[1], attributes ]
431:             end
432:           else
433:             md = @source.match( TEXT_PATTERN, true )
434:             if md[0].length == 0
435:               @source.match( /(\s+)/, true )
436:             end
437:             #STDERR.puts "GOT #{md[1].inspect}" unless md[0].length == 0
438:             #return [ :text, "" ] if md[0].length == 0
439:             # unnormalized = Text::unnormalize( md[1], self )
440:             # return PullEvent.new( :text, md[1], unnormalized )
441:             return [ :text, md[1] ]
442:           end
443:         rescue REXML::UndefinedNamespaceException
444:           raise
445:         rescue REXML::ParseException
446:           raise
447:         rescue Exception, NameError => error
448:           raise REXML::ParseException.new( "Exception parsing",
449:             @source, self, (error ? error : $!) )
450:         end
451:         return [ :dummy ]
452:       end
