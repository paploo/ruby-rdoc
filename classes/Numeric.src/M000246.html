/*
 *  call-seq:
 *     num.abs   => num or numeric
 *
 *  Returns the absolute value of <i>num</i>.
 *
 *     12.abs         #=> 12
 *     (-34.56).abs   #=> 34.56
 *     -34.56.abs     #=> 34.56
 */

static VALUE
num_abs(VALUE num)
{
    if (RTEST(rb_funcall(num, '<', 1, INT2FIX(0)))) {
        return rb_funcall(num, rb_intern("-@"), 0);
    }
    return num;
}
