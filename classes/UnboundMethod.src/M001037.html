static VALUE
method_clone(VALUE self)
{
    VALUE clone;
    struct METHOD *orig, *data;

    Data_Get_Struct(self, struct METHOD, orig);
    clone = Data_Make_Struct(CLASS_OF(self), struct METHOD, bm_mark, -1, data);
    CLONESETUP(clone, self);
    *data = *orig;

    return clone;
}
