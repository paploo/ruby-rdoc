/*
 *  call-seq:
 *     obj.untrust -> obj
 *  
 *  Marks <i>obj</i> as untrusted.
 */

VALUE
rb_obj_untrust(VALUE obj)
{
    rb_secure(4);
    if (!OBJ_UNTRUSTED(obj)) {
        if (OBJ_FROZEN(obj)) {
            rb_error_frozen("object");
        }
        OBJ_UNTRUST(obj);
    }
    return obj;
}
