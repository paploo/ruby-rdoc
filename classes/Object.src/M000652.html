/*
 *  call-seq:
 *     obj.trust    => obj
 *  
 *  Removes the untrusted mark from <i>obj</i>.
 */

VALUE
rb_obj_trust(VALUE obj)
{
    rb_secure(3);
    if (OBJ_UNTRUSTED(obj)) {
        if (OBJ_FROZEN(obj)) {
            rb_error_frozen("object");
        }
        FL_UNSET(obj, FL_UNTRUSTED);
    }
    return obj;
}
