434:     def preprocess(string)
435:       # combine CR+NULL into CR
436:       string = string.gsub(/#{CR}#{NULL}/no, CR) if @options["Telnetmode"]
437: 
438:       # combine EOL into "\n"
439:       string = string.gsub(/#{EOL}/no, "\n") unless @options["Binmode"]
440: 
441:       # remove NULL
442:       string = string.gsub(/#{NULL}/no, '') unless @options["Binmode"]
443: 
444:       string.gsub(/#{IAC}(
445:                    [#{IAC}#{AO}#{AYT}#{DM}#{IP}#{NOP}]|
446:                    [#{DO}#{DONT}#{WILL}#{WONT}]
447:                      [#{OPT_BINARY}-#{OPT_NEW_ENVIRON}#{OPT_EXOPL}]|
448:                    #{SB}[^#{IAC}]*#{IAC}#{SE}
449:                  )/xno) do
450:         if    IAC == $1  # handle escaped IAC characters
451:           IAC
452:         elsif AYT == $1  # respond to "IAC AYT" (are you there)
453:           self.write("nobody here but us pigeons" + EOL)
454:           ''
455:         elsif DO[0] == $1[0]  # respond to "IAC DO x"
456:           if OPT_BINARY[0] == $1[1]
457:             @telnet_option["BINARY"] = true
458:             self.write(IAC + WILL + OPT_BINARY)
459:           else
460:             self.write(IAC + WONT + $1[1..1])
461:           end
462:           ''
463:         elsif DONT[0] == $1[0]  # respond to "IAC DON'T x" with "IAC WON'T x"
464:           self.write(IAC + WONT + $1[1..1])
465:           ''
466:         elsif WILL[0] == $1[0]  # respond to "IAC WILL x"
467:           if    OPT_BINARY[0] == $1[1]
468:             self.write(IAC + DO + OPT_BINARY)
469:           elsif OPT_ECHO[0] == $1[1]
470:             self.write(IAC + DO + OPT_ECHO)
471:           elsif OPT_SGA[0]  == $1[1]
472:             @telnet_option["SGA"] = true
473:             self.write(IAC + DO + OPT_SGA)
474:           else
475:             self.write(IAC + DONT + $1[1..1])
476:           end
477:           ''
478:         elsif WONT[0] == $1[0]  # respond to "IAC WON'T x"
479:           if    OPT_ECHO[0] == $1[1]
480:             self.write(IAC + DONT + OPT_ECHO)
481:           elsif OPT_SGA[0]  == $1[1]
482:             @telnet_option["SGA"] = false
483:             self.write(IAC + DONT + OPT_SGA)
484:           else
485:             self.write(IAC + DONT + $1[1..1])
486:           end
487:           ''
488:         else
489:           ''
490:         end
491:       end
492:     end
