/*
 *  call-seq:
 *     strio.set_encoding(ext_enc)                => strio
 *
 *  Tagged with the encoding specified.
 */

static VALUE
strio_set_encoding(VALUE self, VALUE ext_enc)
{
    rb_encoding* enc;
    VALUE str = StringIO(self)->string;
    enc = rb_to_encoding(ext_enc);
    rb_enc_associate(str, enc);
    return self;
}
