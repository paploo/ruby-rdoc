701:     def find_proxy
702:       name = self.scheme.downcase + '_proxy'
703:       proxy_uri = nil
704:       if name == 'http_proxy' && ENV.include?('REQUEST_METHOD') # CGI?
705:         # HTTP_PROXY conflicts with *_proxy for proxy settings and
706:         # HTTP_* for header information in CGI.
707:         # So it should be careful to use it.
708:         pairs = ENV.reject {|k, v| /\Ahttp_proxy\z/i !~ k }
709:         case pairs.length
710:         when 0 # no proxy setting anyway.
711:           proxy_uri = nil
712:         when 1
713:           k, v = pairs.shift
714:           if k == 'http_proxy' && ENV[k.upcase] == nil
715:             # http_proxy is safe to use because ENV is case sensitive.
716:             proxy_uri = ENV[name]
717:           else
718:             proxy_uri = nil
719:           end
720:         else # http_proxy is safe to use because ENV is case sensitive.
721:           proxy_uri = ENV.to_hash[name]
722:         end
723:         if !proxy_uri
724:           # Use CGI_HTTP_PROXY.  cf. libwww-perl.
725:           proxy_uri = ENV["CGI_#{name.upcase}"]
726:         end
727:       elsif name == 'http_proxy'
728:         unless proxy_uri = ENV[name]
729:           if proxy_uri = ENV[name.upcase]
730:             warn 'The environment variable HTTP_PROXY is discouraged.  Use http_proxy.'
731:           end
732:         end
733:       else
734:         proxy_uri = ENV[name] || ENV[name.upcase]
735:       end
736: 
737:       if proxy_uri && self.host
738:         require 'socket'
739:         begin
740:           addr = IPSocket.getaddress(self.host)
741:           proxy_uri = nil if /\A127\.|\A::1\z/ =~ addr
742:         rescue SocketError
743:         end
744:       end
745: 
746:       if proxy_uri
747:         proxy_uri = URI.parse(proxy_uri)
748:         name = 'no_proxy'
749:         if no_proxy = ENV[name] || ENV[name.upcase]
750:           no_proxy.scan(/([^:,]*)(?::(\d+))?/) {|host, port|
751:             if /(\A|\.)#{Regexp.quote host}\z/i =~ self.host &&
752:                (!port || self.port == port.to_i)
753:               proxy_uri = nil
754:               break
755:             end
756:           }
757:         end
758:         proxy_uri
759:       else
760:         nil
761:       end
762:     end
