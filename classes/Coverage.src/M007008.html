/*
 * call-seq:
 *    Coverage.start  => nil
 *
 * Enables coverage measurement.
 */
static VALUE
rb_coverage_start(VALUE klass)
{
    if (!RTEST(rb_get_coverages())) {
        VALUE coverages = rb_hash_new();
        RBASIC(coverages)->klass = 0;
        rb_set_coverages(coverages);
    }
    return Qnil;
}
