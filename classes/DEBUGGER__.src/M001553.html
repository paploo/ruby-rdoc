220:   def debug_command(file, line, id, binding)
221:     MUTEX.lock
222:     unless defined?($debugger_restart) and $debugger_restart
223:       callcc{|c| $debugger_restart = c} 
224:     end
225:     set_last_thread(Thread.current)
226:     frame_pos = 0
227:     binding_file = file
228:     binding_line = line
229:     previous_line = nil
230:     if ENV['EMACS']
231:       stdout.printf "\032\032%s:%d:\n", binding_file, binding_line
232:     else
233:       stdout.printf "%s:%d:%s", binding_file, binding_line,
234:         line_at(binding_file, binding_line)
235:     end
236:     @frames[0] = [binding, file, line, id]
237:     display_expressions(binding)
238:     prompt = true
239:     while prompt and input = readline("(rdb:%d) "%thnum(), true)
240:       catch(:debug_error) do
241:         if input == ""
242:           next unless DEBUG_LAST_CMD[0]
243:           input = DEBUG_LAST_CMD[0]
244:           stdout.print input, "\n"
245:         else
246:           DEBUG_LAST_CMD[0] = input
247:         end
248: 
249:         case input
250:         when /^\s*tr(?:ace)?(?:\s+(on|off))?(?:\s+(all))?$/
251:           if defined?( $2 )
252:             if $1 == 'on'
253:               set_trace_all true
254:             else
255:               set_trace_all false
256:             end
257:           elsif defined?( $1 )
258:             if $1 == 'on'
259:               set_trace true
260:             else
261:               set_trace false
262:             end
263:           end
264:           if trace?
265:             stdout.print "Trace on.\n"
266:           else
267:             stdout.print "Trace off.\n"
268:           end
269: 
270:         when /^\s*b(?:reak)?\s+(?:(.+):)?([^.:]+)$/
271:           pos = $2
272:           if $1
273:             klass = debug_silent_eval($1, binding)
274:             file = $1
275:           end
276:           if pos =~ /^\d+$/
277:             pname = pos
278:             pos = pos.to_i
279:           else
280:             pname = pos = pos.intern.id2name
281:           end
282:           break_points.push [true, 0, klass || file, pos]
283:           stdout.printf "Set breakpoint %d at %s:%s\n", break_points.size, klass || file, pname
284: 
285:         when /^\s*b(?:reak)?\s+(.+)[#.]([^.:]+)$/
286:           pos = $2.intern.id2name
287:           klass = debug_eval($1, binding)
288:           break_points.push [true, 0, klass, pos]
289:           stdout.printf "Set breakpoint %d at %s.%s\n", break_points.size, klass, pos
290: 
291:         when /^\s*wat(?:ch)?\s+(.+)$/
292:           exp = $1
293:           break_points.push [true, 1, exp]
294:           stdout.printf "Set watchpoint %d:%s\n", break_points.size, exp
295: 
296:         when /^\s*b(?:reak)?$/
297:           if break_points.find{|b| b[1] == 0}
298:             n = 1
299:             stdout.print "Breakpoints:\n"
300:             break_points.each do |b|
301:               if b[0] and b[1] == 0
302:                 stdout.printf "  %d %s:%s\n", n, b[2], b[3] 
303:               end
304:               n += 1
305:             end
306:           end
307:           if break_points.find{|b| b[1] == 1}
308:             n = 1
309:             stdout.print "\n"
310:             stdout.print "Watchpoints:\n"
311:             for b in break_points
312:               if b[0] and b[1] == 1
313:                 stdout.printf "  %d %s\n", n, b[2]
314:               end
315:               n += 1
316:             end
317:           end
318:           if break_points.size == 0
319:             stdout.print "No breakpoints\n"
320:           else
321:             stdout.print "\n"
322:           end
323: 
324:         when /^\s*del(?:ete)?(?:\s+(\d+))?$/
325:           pos = $1
326:           unless pos
327:             input = readline("Clear all breakpoints? (y/n) ", false)
328:             if input == "y"
329:               for b in break_points
330:                 b[0] = false
331:               end
332:             end
333:           else
334:             pos = pos.to_i
335:             if break_points[pos-1]
336:               break_points[pos-1][0] = false
337:             else
338:               stdout.printf "Breakpoint %d is not defined\n", pos
339:             end
340:           end
341: 
342:         when /^\s*disp(?:lay)?\s+(.+)$/
343:           exp = $1
344:           display.push [true, exp]
345:           stdout.printf "%d: ", display.size
346:           display_expression(exp, binding)
347: 
348:         when /^\s*disp(?:lay)?$/
349:           display_expressions(binding)
350: 
351:         when /^\s*undisp(?:lay)?(?:\s+(\d+))?$/
352:           pos = $1
353:           unless pos
354:             input = readline("Clear all expressions? (y/n) ", false)
355:             if input == "y"
356:               for d in display
357:                 d[0] = false
358:               end
359:             end
360:           else
361:             pos = pos.to_i
362:             if display[pos-1]
363:               display[pos-1][0] = false
364:             else
365:               stdout.printf "Display expression %d is not defined\n", pos
366:             end
367:           end
368: 
369:         when /^\s*c(?:ont)?$/
370:           prompt = false
371: 
372:         when /^\s*s(?:tep)?(?:\s+(\d+))?$/
373:           if $1
374:             lev = $1.to_i
375:           else
376:             lev = 1
377:           end
378:           @stop_next = lev
379:           prompt = false
380: 
381:         when /^\s*n(?:ext)?(?:\s+(\d+))?$/
382:           if $1
383:             lev = $1.to_i
384:           else
385:             lev = 1
386:           end
387:           @stop_next = lev
388:           @no_step = @frames.size - frame_pos
389:           prompt = false
390: 
391:         when /^\s*w(?:here)?$/, /^\s*f(?:rame)?$/
392:           display_frames(frame_pos)
393: 
394:         when /^\s*l(?:ist)?(?:\s+(.+))?$/
395:           if not $1
396:             b = previous_line ? previous_line + 10 : binding_line - 5
397:             e = b + 9
398:           elsif $1 == '-'
399:             b = previous_line ? previous_line - 10 : binding_line - 5
400:             e = b + 9
401:           else
402:             b, e = $1.split(/[-,]/)
403:             if e
404:               b = b.to_i
405:               e = e.to_i
406:             else
407:               b = b.to_i - 5
408:               e = b + 9
409:             end
410:           end
411:           previous_line = b
412:           display_list(b, e, binding_file, binding_line)
413: 
414:         when /^\s*up(?:\s+(\d+))?$/
415:           previous_line = nil
416:           if $1
417:             lev = $1.to_i
418:           else
419:             lev = 1
420:           end
421:           frame_pos += lev
422:           if frame_pos >= @frames.size
423:             frame_pos = @frames.size - 1
424:             stdout.print "At toplevel\n"
425:           end
426:           binding, binding_file, binding_line = @frames[frame_pos]
427:           stdout.print format_frame(frame_pos)
428: 
429:         when /^\s*down(?:\s+(\d+))?$/
430:           previous_line = nil
431:           if $1
432:             lev = $1.to_i
433:           else
434:             lev = 1
435:           end
436:           frame_pos -= lev
437:           if frame_pos < 0
438:             frame_pos = 0
439:             stdout.print "At stack bottom\n"
440:           end
441:           binding, binding_file, binding_line = @frames[frame_pos]
442:           stdout.print format_frame(frame_pos)
443: 
444:         when /^\s*fin(?:ish)?$/
445:           if frame_pos == @frames.size
446:             stdout.print "\"finish\" not meaningful in the outermost frame.\n"
447:           else
448:             @finish_pos = @frames.size - frame_pos
449:             frame_pos = 0
450:             prompt = false
451:           end
452: 
453:         when /^\s*cat(?:ch)?(?:\s+(.+))?$/
454:           if $1
455:             excn = $1
456:             if excn == 'off'
457:               @catch = nil
458:               stdout.print "Clear catchpoint.\n"
459:             else
460:               @catch = excn
461:               stdout.printf "Set catchpoint %s.\n", @catch
462:             end
463:           else
464:             if @catch
465:               stdout.printf "Catchpoint %s.\n", @catch
466:             else
467:               stdout.print "No catchpoint.\n"
468:             end
469:           end
470: 
471:         when /^\s*q(?:uit)?$/
472:           input = readline("Really quit? (y/n) ", false)
473:           if input == "y"
474:             exit!      # exit -> exit!: No graceful way to stop threads...
475:           end
476: 
477:         when /^\s*v(?:ar)?\s+/
478:           debug_variable_info($', binding)
479: 
480:         when /^\s*m(?:ethod)?\s+/
481:           debug_method_info($', binding)
482: 
483:         when /^\s*th(?:read)?\s+/
484:           if DEBUGGER__.debug_thread_info($', binding) == :cont
485:             prompt = false
486:           end
487: 
488:         when /^\s*pp\s+/
489:           PP.pp(debug_eval($', binding), stdout)
490: 
491:         when /^\s*p\s+/
492:           stdout.printf "%s\n", debug_eval($', binding).inspect
493: 
494:         when /^\s*r(?:estart)?$/
495:           $debugger_restart.call
496: 
497:         when /^\s*h(?:elp)?$/
498:           debug_print_help()
499: 
500:         else
501:           v = debug_eval(input, binding)
502:           stdout.printf "%s\n", v.inspect
503:         end
504:       end
505:     end
506:     MUTEX.unlock
507:     resume_all
508:   end
