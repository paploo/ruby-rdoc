312:           end
313:           if (@canonical_names.include?(i))
314:             raise ArgumentError, "option redefined `#{i}'"
315:           end
316:         rescue
317:           @canonical_names.clear
318:           @argument_flags.clear
319:           raise
320:         end
321:

324:         # `@canonical_names' Hashes.

328:         end
329:         @canonical_names[i] = canonical_name
