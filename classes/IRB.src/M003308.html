128:   def IRB.parse_opts
129:     load_path = []
130:     while opt = ARGV.shift
131:       case opt
132:       when "-f"
133:         @CONF[:RC] = false
134:       when "-m"
135:         @CONF[:MATH_MODE] = true
136:       when "-d"
137:         $DEBUG = true
138:       when /^-r(.+)?/
139:         opt = $1 || ARGV.shift
140:         @CONF[:LOAD_MODULES].push opt if opt
141:       when /^-I(.+)?/
142:         opt = $1 || ARGV.shift
143:         load_path.concat(opt.split(File::PATH_SEPARATOR)) if opt
144:       when '-U'
145:         set_encoding("UTF-8", "UTF-8")
146:       when /^-E(.+)?/, /^--encoding(?:=(.+))?/
147:         opt = $1 || ARGV.shift
148:         set_encoding(*opt.split(':', 2))
149:       when "--inspect"
150:         @CONF[:INSPECT_MODE] = true
151:       when "--noinspect"
152:         @CONF[:INSPECT_MODE] = false
153:       when "--readline"
154:         @CONF[:USE_READLINE] = true
155:       when "--noreadline"
156:         @CONF[:USE_READLINE] = false
157:       when "--echo"
158:         @CONF[:ECHO] = true
159:       when "--noecho"
160:         @CONF[:ECHO] = false
161:       when "--verbose"
162:         @CONF[:VERBOSE] = true
163:       when "--noverbose"
164:         @CONF[:VERBOSE] = false
165:       when /^--prompt-mode(?:=(.+))?/, /^--prompt(?:=(.+))?/
166:         opt = $1 || ARGV.shift
167:         prompt_mode = opt.upcase.tr("-", "_").intern
168:         @CONF[:PROMPT_MODE] = prompt_mode
169:       when "--noprompt"
170:         @CONF[:PROMPT_MODE] = :NULL
171:       when "--inf-ruby-mode"
172:         @CONF[:PROMPT_MODE] = :INF_RUBY
173:       when "--sample-book-mode", "--simple-prompt"
174:         @CONF[:PROMPT_MODE] = :SIMPLE
175:       when "--tracer"
176:         @CONF[:USE_TRACER] = true
177:       when /^--back-trace-limit(?:=(.+))?/
178:         @CONF[:BACK_TRACE_LIMIT] = ($1 || ARGV.shift).to_i
179:       when /^--context-mode(?:=(.+))?/
180:         @CONF[:CONTEXT_MODE] = ($1 || ARGV.shift).to_i
181:       when "--single-irb"
182:         @CONF[:SINGLE_IRB] = true
183:       when /^--irb_debug=(?:=(.+))?/
184:         @CONF[:DEBUG_LEVEL] = ($1 || ARGV.shift).to_i
185:       when "-v", "--version"
186:         print IRB.version, "\n"
187:         exit 0
188:       when "-h", "--help"
189:         require "irb/help"
190:         IRB.print_usage
191:         exit 0
192:       when "--"
193:         if opt = ARGV.shfit
194:           @CONF[:SCRIPT] = opt
195:           $0 = opt
196:         end
197:         break
198:       when /^-/
199:         IRB.fail UnrecognizedSwitch, opt
200:       else
201:         @CONF[:SCRIPT] = opt
202:         $0 = opt
203:         break
204:       end
205:     end
206:     if RUBY_VERSION >= FEATURE_IOPT_CHANGE_VERSION
207:       load_path.collect! do |path|
208:         /\A\.\// =~ path ? path : File.expand_path(path)
209:       end
210:     end
211:     $LOAD_PATH.unshift(*load_path)
212: 
213:   end
