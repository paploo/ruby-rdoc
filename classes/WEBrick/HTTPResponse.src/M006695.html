257:     def send_body_io(socket)
258:       begin
259:         if @request_method == "HEAD"
260:           # do nothing
261:         elsif chunked?
262:           while buf = @body.read(@buffer_size)
263:             next if buf.empty?
264:             data = ""
265:             data << format("%x", buf.bytesize) << CRLF
266:             data << buf << CRLF
267:             _write_data(socket, data)
268:             @sent_size += buf.bytesize
269:           end
270:           _write_data(socket, "0#{CRLF}#{CRLF}")
271:         else
272:           size = @header['content-length'].to_i
273:           _send_file(socket, @body, 0, size)
274:           @sent_size = size
275:         end
276:       ensure
277:         @body.close
278:       end
279:     end
