 85:     def do_CONNECT(req, res)
 86:       # Proxy Authentication
 87:       proxy_auth(req, res)
 88: 
 89:       ua = Thread.current[:WEBrickSocket]  # User-Agent
 90:       raise HTTPStatus::InternalServerError,
 91:         "[BUG] cannot get socket" unless ua
 92: 
 93:       host, port = req.unparsed_uri.split(":", 2)
 94:       # Proxy authentication for upstream proxy server
 95:       if proxy = proxy_uri(req, res)
 96:         proxy_request_line = "CONNECT #{host}:#{port} HTTP/1.0"
 97:         if proxy.userinfo
 98:           credentials = "Basic " + [proxy.userinfo].pack("m").delete("\n")
 99:         end
100:         host, port = proxy.host, proxy.port
101:       end
102: 
103:       begin
104:         @logger.debug("CONNECT: upstream proxy is `#{host}:#{port}'.")
105:         os = TCPSocket.new(host, port)     # origin server
106: 
107:         if proxy
108:           @logger.debug("CONNECT: sending a Request-Line")
109:           os << proxy_request_line << CRLF
110:           @logger.debug("CONNECT: > #{proxy_request_line}")
111:           if credentials
112:             @logger.debug("CONNECT: sending a credentials")
113:             os << "Proxy-Authorization: " << credentials << CRLF
114:           end
115:           os << CRLF
116:           proxy_status_line = os.gets(LF)
117:           @logger.debug("CONNECT: read a Status-Line form the upstream server")
118:           @logger.debug("CONNECT: < #{proxy_status_line}")
119:           if %r{^HTTP/\d+\.\d+\s+200\s*} =~ proxy_status_line
120:             while line = os.gets(LF)
121:               break if /\A(#{CRLF}|#{LF})\z/om =~ line
122:             end
123:           else
124:             raise HTTPStatus::BadGateway
125:           end
126:         end
127:         @logger.debug("CONNECT #{host}:#{port}: succeeded")
128:         res.status = HTTPStatus::RC_OK
129:       rescue => ex
130:         @logger.debug("CONNECT #{host}:#{port}: failed `#{ex.message}'")
131:         res.set_error(ex)
132:         raise HTTPStatus::EOFError
133:       ensure
134:         if handler = @config[:ProxyContentHandler]
135:           handler.call(req, res)
136:         end
137:         res.send_response(ua)
138:         access_log(@config, req, res)
139: 
140:         # Should clear request-line not to send the sesponse twice.
141:         # see: HTTPServer#run
142:         req.parse(NullReader) rescue nil
143:       end
144: 
145:       begin
146:         while fds = IO::select([ua, os])
147:           if fds[0].member?(ua)
148:             buf = ua.sysread(1024);
149:             @logger.debug("CONNECT: #{buf.bytesize} byte from User-Agent")
150:             os.syswrite(buf)
151:           elsif fds[0].member?(os)
152:             buf = os.sysread(1024);
153:             @logger.debug("CONNECT: #{buf.bytesize} byte from #{host}:#{port}")
154:             ua.syswrite(buf)
155:           end
156:         end
157:       rescue => ex
158:         os.close
159:         @logger.debug("CONNECT #{host}:#{port}: closed")
160:       end
161: 
162:       raise HTTPStatus::EOFError
163:     end
