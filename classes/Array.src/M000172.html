/*
 *  call-seq:
 *     array.shuffle!        -> array
 *  
 *  Shuffles elements in _self_ in place.
 */


static VALUE
rb_ary_shuffle_bang(VALUE ary)
{
    long i = RARRAY_LEN(ary);

    rb_ary_modify(ary);
    while (i) {
        long j = rb_genrand_real()*i;
        VALUE tmp = RARRAY_PTR(ary)[--i];
        RARRAY_PTR(ary)[i] = RARRAY_PTR(ary)[j];
        RARRAY_PTR(ary)[j] = tmp;
    }
    return ary;
}
