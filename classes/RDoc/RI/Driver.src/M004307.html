514:   def lookup_ancestor(klass, orig_klass)
515:     # This is a bit hacky, but ri will go into an infinite
516:     # loop otherwise, since Object has an Object ancestor
517:     # for some reason.  Depending on the documentation state, I've seen
518:     # Kernel as an ancestor of Object and not as an ancestor of Object.
519:     if ((orig_klass == "Object") &&
520:         ((klass == "Kernel") || (klass == "Object")))
521:       return nil
522:     end
523: 
524:     cache = class_cache[orig_klass]
525: 
526:     return nil unless cache
527: 
528:     ancestors = [orig_klass]
529:     ancestors.push(*cache.includes.map { |inc| inc['name'] })
530:     ancestors << cache.superclass
531:     
532:     ancestor_index = ancestors.index(klass)
533: 
534:     if ancestor_index
535:       ancestor = ancestors[ancestors.index(klass) + 1]
536:       return ancestor if ancestor
537:     end
538: 
539:     lookup_ancestor klass, cache.superclass
540:   end
