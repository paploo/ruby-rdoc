181:   def display_class_method_list(klass)
182:     method_map = {}
183: 
184:     class_data = [
185:                   :class_methods,
186:                   :class_method_extensions,
187:                   :instance_methods,
188:                   :instance_method_extensions,
189:                  ]
190:     
191:     class_data.each do |data_type|
192:       data = klass.send data_type
193:       
194:       unless data.nil? or data.empty? then
195:         @formatter.blankline
196:         
197:         heading = data_type.to_s.split('_').join(' ').capitalize << ':'
198:         @formatter.display_heading heading, 2, ''
199:         
200:         method_names = []
201:         data.each do |item|
202:           method_names << item.name
203: 
204:           if(data_type == :class_methods ||
205:              data_type == :class_method_extensions) then
206:             method_map["::#{item.name}"] = :class
207:             method_map[item.name] = :class
208:           else
209:             #
210:             # Since we iterate over instance methods after class methods,
211:             # an instance method always will overwrite the unqualified
212:             # class method entry for a class method of the same name.
213:             #
214:             method_map["##{item.name}"] = :instance
215:             method_map[item.name] = :instance
216:           end
217:         end
218:         method_names.sort!
219: 
220:         @formatter.wrap method_names.join(', ')
221:       end
222:     end
223: 
224:     method_map
225:   end
