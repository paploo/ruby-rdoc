146:     def normalized_file_list(options, relative_files, force_doc = false,
147:                              exclude_pattern = nil)
148:       file_list = []
149: 
150:       relative_files.each do |rel_file_name|
151:         next if exclude_pattern && exclude_pattern =~ rel_file_name
152:         stat = File.stat(rel_file_name)
153:         case type = stat.ftype
154:         when "file"
155:           next if @last_created and stat.mtime < @last_created
156: 
157:           if force_doc or ::RDoc::Parser.can_parse(rel_file_name) then
158:             file_list << rel_file_name.sub(/^\.\//, '')
159:           end
160:         when "directory"
161:           next if rel_file_name == "CVS" || rel_file_name == ".svn"
162:           dot_doc = File.join(rel_file_name, DOT_DOC_FILENAME)
163:           if File.file?(dot_doc)
164:             file_list.concat(parse_dot_doc_file(rel_file_name, dot_doc, options))
165:           else
166:             file_list.concat(list_files_in_directory(rel_file_name, options))
167:           end
168:         else
169:           raise RDoc::Error, "I can't deal with a #{type} #{rel_file_name}"
170:         end
171:       end
172: 
173:       file_list
174:     end
