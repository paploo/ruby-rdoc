/*
 *  call-seq:
 *     Math.atanh(x)    => float
 *  
 *  Computes the inverse hyperbolic tangent of <i>x</i>.
 */

static VALUE
math_atanh(VALUE obj, VALUE x)
{
    double d0, d;

    Need_Float(x);
    errno = 0;
    d0 = RFLOAT_VALUE(x);
    d = atanh(d0);
    domain_check(d0, d, "atanh");
    infinity_check(x, d, "atanh");
    return DBL2NUM(d);
}
