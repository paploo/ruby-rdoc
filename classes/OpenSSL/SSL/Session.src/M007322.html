/*
 * call-seq:
 *    session.timeout -> integer
 *
 * How long until the session expires in seconds.
 *
*/
static VALUE ossl_ssl_session_get_timeout(VALUE self)
{
        SSL_SESSION *ctx;
        time_t t;

        GetSSLSession(self, ctx);

        t = SSL_SESSION_get_timeout(ctx);

        return ULONG2NUM(t);
}
