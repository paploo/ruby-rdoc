/*  call-seq:
 *     group.asn1_flag  => Fixnum
 *
 *  See the OpenSSL documentation for EC_GROUP_get_asn1_flag()
 */
static VALUE ossl_ec_group_get_asn1_flag(VALUE self)
{
    EC_GROUP *group = NULL;
    int flag;

    Require_EC_GROUP(self, group);

    flag = EC_GROUP_get_asn1_flag(group);

    return INT2FIX(flag);
}
