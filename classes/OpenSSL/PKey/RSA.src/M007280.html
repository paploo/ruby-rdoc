/*
 *  call-seq:
 *     rsa.public? -> true
 *
 *  The return value is always true since every private key is also a public key.
 *
 */
static VALUE
ossl_rsa_is_public(VALUE self)
{
    EVP_PKEY *pkey;

    GetPKeyRSA(self, pkey);
    /*
     * This method should check for n and e.  BUG.
     */
    return Qtrue;
}
